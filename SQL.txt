create table Administrador(
id_adm INTEGER NOT NULL PRIMARY KEY,
login VARCHAR2(30) NOT NULL,
senha VARCHAR2(50) NOT NULL
);

insert into Administrador (id_adm, login, senha) values (20, 'admin', 'admin');

create table Especialidade(
id_especialidade INTEGER NOT NULL PRIMARY KEY,
tipo_especialidade VARCHAR2(30) not null,
pagamento NUMBER(10,2) not null
);

create table Medico(
id_crm VARCHAR2(30) NOT NULL PRIMARY KEY,
nome VARCHAR2(40) NOT NULL,
Administrador_id_adm INTEGER,
Especialidade_id_especialidade INTEGER,
FOREIGN KEY(Administrador_id_adm) REFERENCES Administrador(id_adm),
FOREIGN KEY(Especialidade_id_especialidade) REFERENCES Especialidade(id_especialidade)
);

create table Paciente(
id_paciente INTEGER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
nome VARCHAR2(40),
email VARCHAR2(40),
cpf VARCHAR2(11),
login VARCHAR2(50),
senha VARCHAR2(70),
Administrador_id_adm INTEGER,
FOREIGN KEY (Administrador_id_adm) REFERENCES Administrador(id_adm)
);

create table Consulta(
id_consulta integer not null PRIMARY KEY,
agendamento DATE,
Paciente_id_paciente INTEGER,
Medico_id_crm VARCHAR2(30),
FOREIGN KEY (Paciente_id_paciente) REFERENCES Paciente(id_paciente),
FOREIGN KEY (Medico_id_crm) REFERENCES Medico(id_crm)
);

COMMIT;
